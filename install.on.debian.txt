# install dependencies
  apt -y install apt-transport-https ca-certificates curl gnupg2

# Add Docker repository key
  curl -fsSL https://download.docker.com/linux/debian/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

# Add Docker APT repository (change amd64 to arm64 if needed)
  echo "deb [arch=amd64 signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/debian $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list

# Install Docker and docker-compose
  update.system.sh
  apt-get -y install docker-ce docker-ce-cli containerd.io docker-compose
  docker --version

# Check list of Docker containers
  docker ps
  docker ps -a

# Hello World Examples
  docker run hello-world
  docker run -d -p 80:80 nginx
  curl localhost
  docker ps
  docker stop <ID>
  docker container prune

# Networking - Enable container to get on same network as docker
docker network create -d macvlan \
  --subnet=192.168.1.0/24 \
  --gateway=192.168.1.1 \
  -o parent=enX0 \
  macvlan_lan

docker run --rm -it --network=macvlan_lan alpine ash
